interface "Register<T>" {
    + add(T toAdd) : void
    + modify(T old, T newObj) : void
    + remove(T toRemove) : void
    + isUnique(T toVerify) : boolean
    + getRegisterList() : List<T>
}

interface "RegisterAdder<T>" {
    + add(T toAdd) : void
}

interface "RegisterModifier<T>" {
    + modify(T old, T newObj) : void
}

interface "RegisterValidator<T>" {
    + isUnique(T toVerify) : boolean
}
