@startuml
'DIAGRAMMA DI SEQUENZA DELLA MODIFICA DI UN PRESTITO

actor Admin

participant "<u>:LoanRegisterController" as LoanRegisterController
participant "<u>:LoanModifyPopupController" as LoanModifyPopupController
participant "<u>:LoanRegister" as LoanRegister

Admin -> LoanRegisterController : "Selects a Loan"
Admin -> LoanRegisterController : openModifyPopup()
LoanRegisterController --> Admin : "Shows Modify Popup"
loop LoanNotValid
Admin -> LoanModifyPopupController : confirmBtnAction(event)
opt LoanNotValid
LoanModifyPopupController --> Admin : "Shows Error Alert"
end
end
LoanModifyPopupController -> LoanRegister : modify(selectedLoan, newLoan)
@enduml
