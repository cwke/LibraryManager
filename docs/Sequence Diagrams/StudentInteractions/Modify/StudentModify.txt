@startuml
'DIAGRAMMA DI SEQUENZA DELLA MODIFICA DI UNO STUDENTE

actor Admin

participant "<u>:StudentRegisterController" as StudentRegisterController
participant "<u>:StudentModifyPopupController" as StudentModifyPopupController
participant "<u>:StudentRegister" as StudentRegister

Admin -> StudentRegisterController : "Selects a Student"
Admin -> StudentRegisterController : openModifyPopup()
StudentRegisterController --> Admin : "Shows Modify Popup"
loop StudentNotValid
Admin -> StudentModifyPopupController : confirmBtnAction(event)
opt StudentNotValid
StudentModifyPopupController --> Admin : "Shows Error Alert"
end
end
StudentModifyPopupController -> StudentRegister : modify(selectedStudent, newStudent)
@enduml
