@startuml
'DIAGRAMMA DI SEQUENZA DELLA MODIFICA DI UN LIBRO

actor Admin

participant "<u>:BookRegisterController" as BookRegisterController
participant "<u>:BookModifyPopupController" as BookModifyPopupController
participant "<u>:BookRegister" as BookRegister

Admin -> BookRegisterController : "Selects a Book"
Admin -> BookRegisterController : openModifyPopup()
BookRegisterController --> Admin : "Shows Modify Popup"
loop BookNotValid
Admin -> BookModifyPopupController : confirmBtnAction(event)
opt BookNotValid
BookModifyPopupController --> Admin : "Shows Error Alert"
end
end
BookModifyPopupController -> BookRegister : modify(selectedBook, newBook)
@enduml
