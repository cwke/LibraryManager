@startuml
'DIAGRAMMA DI SEQUENZA DELL'INSERIMENTO DI UN LIBRO

actor Admin

participant "<u>:BookRegisterController" as BookRegisterController
participant "<u>:BookInsertPopupController" as BookInsertPopupController
participant "<u>:BookRegister" as BookRegister

Admin -> BookRegisterController : openInsertPopup()
BookRegisterController --> Admin : "Shows Insert Popup"
loop BookNotValid
Admin -> BookInsertPopupController : confirmBtnAction(event)
BookInsertPopupController -> BookRegister : isUnique(insertedBook)
BookInsertPopupController <-- BookRegister : result
opt BookNotValid
BookInsertPopupController --> Admin : "Shows Error Alert"
end
end
BookInsertPopupController -> BookRegister : add(insertedBook)
@enduml
